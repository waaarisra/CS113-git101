#include <stdio.h>
int addTen(int *number, int n)
{
    for (int i = 0; i < n; i++)
    {
        printf("[%p] = %d\n",number + i, *(number+i));
        *(number + i)+=10;
    }
    return 1;
}
void addTwenty(int *n)//ชื้ ไปที่ค่าใน n 
{
    *n +=20;
}
int main()
{
    int numbers[] = {4,5,6,7,8};
    int n = 5;
    addTen(numbers,n); 
    for (int i = 0; i < n; i++)
    {
        printf("main [%p] = %d\n",numbers + i, *(numbers +i));//%p ปริ้น address เลขฐาน 16,%d ปริ้นค่าที่ตำแหน่ง numbers + i 
    }
    addTwenty(&n); //address ของ n 
    printf("n = %d\n",n);
}
